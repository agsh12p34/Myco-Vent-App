<script>
document.addEventListener("DOMContentLoaded", () => {
    // 1. RATING LOGIC
    const ratingButtons = document.querySelectorAll('.rating-btn');
    ratingButtons.forEach(btn => {
        btn.addEventListener('click', (e) => {
            const val = parseInt(e.target.getAttribute('data-val'));
            ratingButtons.forEach(b => b.classList.remove('ring-2', 'ring-offset-1'));
            e.target.classList.add('ring-2', 'ring-offset-1');

            const aiHub = document.getElementById('aiInputHub');
            const resultsBox = document.getElementById('aiResultsBox');
            resultsBox.classList.add('hidden');

            if (val < 3) {
                aiHub.classList.remove('hidden');
            } else {
                aiHub.classList.add('hidden');
            }
        });
    });

    // 2. AUDIO RECORDING SIMULATION
    const recordBtn = document.getElementById('recordBtn');
    if(recordBtn) {
        recordBtn.addEventListener('click', () => {
            const text = document.getElementById('recordText');
            const icon = document.getElementById('micIcon');
            recordBtn.classList.add('bg-red-50', 'text-red-600', 'border-red-200');
            icon.classList.add('pulse-animation');
            text.innerText = "3s...";

            setTimeout(() => {
                recordBtn.classList.remove('bg-red-50', 'text-red-600', 'border-red-200');
                recordBtn.classList.add('bg-green-50', 'text-green-700', 'border-green-200');
                icon.classList.remove('pulse-animation');
                text.innerText = "Saved";
            }, 3000);
        });
    }

    // 3. AUDIO FILE UPLOAD
    const audioUpload = document.getElementById('audioUpload');
    if(audioUpload) {
        audioUpload.addEventListener('change', (event) => {
            const file = event.target.files[0];
            if (file) {
                const btn = document.getElementById('uploadBtn');
                const text = document.getElementById('uploadText');
                btn.classList.add('bg-green-50', 'text-green-700', 'border-green-300');
                text.innerText = "Attached";
            }
        });
    }

    // 4. PRESENTATION MODE (MOCK AI)
    const submitBtn = document.getElementById('submitAIBtn');
    if(submitBtn) {
        submitBtn.addEventListener('click', () => {
            const submitText = document.getElementById('submitText');
            const resultsBox = document.getElementById('aiResultsBox');
            const patientInput = document.getElementById('symptomDetails').value;
            
            if (!patientInput.trim()) {
                alert("Please enter some symptoms first!");
                return;
            }

            submitBtn.classList.add('opacity-75', 'cursor-not-allowed');
            submitText.innerText = "AI Processing...";

            setTimeout(() => {
                const aiResponse = {
                    summary: "Patient experienced an acute exacerbation. Acoustic analysis confirms sub-audible wheezing matching nocturnal triggers.",
                    rooms: [
                        { name: "Main Bedroom", severity: "High Trigger", score: 85 },
                        { name: "Kitchen", severity: "Mod Trigger", score: 45 }
                    ]
                };
                
                document.getElementById('llmSummaryText').innerText = aiResponse.summary;
                
                let heatmapHTML = '';
                aiResponse.rooms.forEach(room => {
                    let colorClass = room.score > 70 ? 'red' : (room.score > 40 ? 'orange' : 'yellow');
                    heatmapHTML += `
                    <div class="flex items-center justify-between text-xs mt-3">
                        <span class="font-semibold text-${colorClass}-700">${room.name}</span>
                        <span class="text-${colorClass}-500 font-bold">${room.severity}</span>
                    </div>
                    <div class="w-full bg-gray-200 rounded-full h-1.5 mt-1">
                        <div class="bg-${colorClass}-500 h-1.5 rounded-full" style="width: ${room.score}%"></div>
                    </div>`;
                });
                
                document.getElementById('llmHeatmapContainer').innerHTML = heatmapHTML;

                submitBtn.classList.remove('opacity-75', 'cursor-not-allowed');
                submitText.innerText = "Analyze My Symptoms";
                resultsBox.classList.remove('hidden');
            }, 1500); 
        });
    }

    // 5. CHART.JS SETUP
    try {
        const ctx = document.getElementById('evaluationChart').getContext('2d');
        new Chart(ctx, {
            type: 'line',
            data: {
                labels: ['Wk 1', 'Wk 2', 'Wk 3', 'Wk 4'],
                datasets: [
                    { label: 'Humidity (%)', data: [72, 65, 54, 50], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.1)', tension: 0.4, fill: true, yAxisID: 'y' },
                    { label: 'Symptom Score', data: [2, 3, 4, 5], borderColor: '#10b981', borderWidth: 2, borderDash: [5, 5], tension: 0.4, yAxisID: 'y1' }
                ]
            },
            options: {
                responsive: true, maintainAspectRatio: false,
                plugins: { legend: { display: true, position: 'top', labels: { boxWidth: 10, usePointStyle: true, font: { size: 10 } } } },
                scales: {
                    y: { type: 'linear', display: true, position: 'left', min: 40, max: 80, grid: { drawBorder: false, color: '#f3f4f6' }, ticks: { font: { size: 9 } } },
                    y1: { type: 'linear', display: true, position: 'right', min: 1, max: 5, grid: { drawOnChartArea: false }, ticks: { font: { size: 9 } } },
                    x: { grid: { display: false }, ticks: { font: { size: 9 } } }
                }
            }
        });
    } catch (e) {
        console.warn("Chart.js loading delay.");
    }
});
</script>
